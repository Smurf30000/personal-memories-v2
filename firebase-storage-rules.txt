// Firebase Storage Security Rules
// Copy these rules to your Firebase Console: Storage -> Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function to check if user owns the file
    function isOwner(userId) {
      return request.auth.uid == userId;
    }
    
    // Users can only access their own media files
    match /users/{userId}/media/{mediaFile} {
      // Allow read if authenticated and owner
      allow read: if isAuthenticated() && isOwner(userId);
      
      // Allow write if authenticated and owner
      allow write: if isAuthenticated() && isOwner(userId);
      
      // Allow delete if authenticated and owner
      allow delete: if isAuthenticated() && isOwner(userId);
    }
    
    // Deny all other paths by default
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
